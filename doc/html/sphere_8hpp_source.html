127  // Clever trick here: we only need to do half
128  // -> the two sides of the square will give same answer anyways
129  T a = r.direction().norm_sqr();

131  T c = oc.norm_sqr() - r_ * r_;

133  T discriminant = b * b - a * c;
134  if (discriminant < 0)
135  return false;

137 
138  // Find best soln in our time

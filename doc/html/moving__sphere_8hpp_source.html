142  vec3<T> oc = r.origin() - center(r.time());
143 
144  // We'll solve (P-C)\dot(P-C) = r^2
145  // Clever trick here: we only need to do half
146  // -> the two sides of the square will give same answer anyways
147  T a = r.direction().norm_sqr();
148  T b = dot(oc, r.direction());
149  T c = oc.norm_sqr() - r_ * r_;
150 
151  T discriminant = b * b - a * c;
152  if (discriminant < 0)
153  return false;
154  T sqrtd(std::sqrt(discriminant));
155 
156  // Find best soln in our time
157  T soln = (-b - sqrtd) / a;
158  if (soln < t_min || t_max < soln) {
159  soln = (-b + sqrtd) / a;
160  if (soln < t_min || t_max < soln)
161  return false;
162  }
